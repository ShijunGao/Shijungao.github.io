嵌套子查询

  • 独立子查询
    SELECT id
FROM inventory
WHERE name=(SELECT name
            FROM inventory
            WHERE name='Apple');
  • 相关子查询
    SELECT id
FROM inventory e             //别名 e
WHERE name=(SELECT name
          FROM inventory
          WHERE e.name='Apple');
